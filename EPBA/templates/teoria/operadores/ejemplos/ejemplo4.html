Ejemplo #4

Enunciado


Realizar un programa que pida al usuario dos edades y que este calcule la suma del doble de las edades. Imprimir en el monitor serial el resultado obtenido.

Empezar

Solución


                  
 1int edad1;                    
 2int edad2;
 3int suma;
 4void setup(){
 5    Serial.begin(9600);
 6}
 7void loop(){
 8    Serial.print("Ingrese una edad");
 9    edad1 = Serial.read();
10    Serial.print("Ingrese otra edad");
11    edad2 = Serial.read();
12    suma = (edad1*2)+(edad2*2);
13    Serial.print("El resultado es: ");
14    Serial.println(suma);
15}
