Ejemplo #6

Enunciado


Pedir al usuario que ingrese dos números por teclado, luego denotar si la resta del primer número menos el segundo número da como resultado un número negativo, de ser así imprimir true, de lo contrario imprimir false.

Empezar

Solución


                  
 1int numero1;                    
 2int numero2;
 3int resta;
 4void setup(){
 5    Serial.begin(9600);
 6}
 7void loop(){
 8    Serial.print("Ingrese un número");
 9    numero1 = Serial.read();
10    Serial.print("Ingrese otro número");
11    numero2 = Serial.read();
12    resta = numero1-numero2;
13    Serial.print("La expresión da como resultado: ");
14    Serial.println(resta < 0);
15}
