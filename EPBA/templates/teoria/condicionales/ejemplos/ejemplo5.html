Ejemplo #5

Enunciado


Pedir al usuario que ingrese por teclado 3 números enteros n1, n2, n3. Se desea realizar un programa que imprima por pantalla los números ingresados por el usuario de manera ascendente, es decir, de menor a mayor.

Empezar

Solución


                  
 1int numero1;                    
 2int numero2;
 3int numero3;
 4void setup(){
 5    Serial.begin(9600);
 6}
 7void loop(){
 8    Serial.print("Ingrese el primer número");
 9    numero1 = Serial.read();
10    Serial.print("Ingrese el segundo número");
11    numero2 = Serial.read();
12    Serial.print("Ingrese el tercer número");
13    numero3 = Serial.read();
14    if(numero1 < numero2 && numero2 < numero3){
15        if(numero2 < numero3){
16            Serial.println(numero1);
17            Serial.println(numero2);
18            Serial.println(numero3);
19        }else{
20            Serial.println(numero1);
21            Serial.println(numero3);
22            Serial.println(numero2);
23        }
24    }
25    else if(numero2 < numero1 && numero2 < numero3){
26        if(numero1 < numero3){
27            Serial.println(numero2);
28            Serial.println(numero1);
29            Serial.println(numero3);
30        }else{
31            Serial.println(numero2);
32            Serial.println(numero3);
33            Serial.println(numero1);
34        }
35    }
36    else if(numero3 < numero1 && numero3 < numero2){
37        if(numero1 < numero2){
38            Serial.println(numero3);
39            Serial.println(numero1);
40            Serial.println(numero2);
41        }else{
42            Serial.println(numero3);
43            Serial.println(numero2);
44            Serial.println(numero1);
45        }
46    }
47}
