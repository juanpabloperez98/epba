Ejemplo #1

Enunciado


Pida al usuario que ingrese 3 números enteros numero1, numero2 y numero3, lo que el programa debe hacer es determinar cuál es el mayor de los 3 números ingresados por el usuario.

Empezar

Solución


                  
 1int numero1;                    
 2int numero2;
 3int numero3;
 4int mayor;
 5void setup(){
 6    Serial.begin(9600);
 7}
 8void loop(){
 9    Serial.print("Ingrese un número");
10    numero1 = Serial.read();
11    Serial.print("Ingrese otro número");
12    numero2 = Serial.read();
13    Serial.print("Ingrese otro número");
14    numero3 = Serial.read();
15    mayor = numero1;
16    if(mayor < numero2){
17        mayor = numero2;
18    }
19    if(mayor < numero3){
20        mayor = numero3;
21    }
22    Serial.print("El número mayor es: ");
23    Serial.println(mayor);
24}
