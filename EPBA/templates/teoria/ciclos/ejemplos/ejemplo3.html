Ejemplo #3

Enunciado


Realizar un programa que me muestre los números pares comprendidos entre 1 y el número que ingrese el usuario, y que imprima el total de números pares encontrados.

Empezar

Solución


                  
 1int n;                    
 2int pares;
 3void setup(){
 4    Serial.begin(9600);
 5}
 6void loop(){
 7    Serial.print("Ingrese un número: ");
 8    n = Serial.read();
 9    for(int i=1; i<=n; i++){
10        if(i%2 == 0){
11            Serial.print("El número ");
12            Serial.println(i);
13            Serial.print(" es par.");
14            pares = pares + 1;
15        }
16        Serial.print("El número de pares encontrados son: ");
17        Serial.println(pares);
18    }
19}
