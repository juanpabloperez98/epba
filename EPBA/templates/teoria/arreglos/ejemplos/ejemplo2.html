Ejemplo #2

Enunciado

Escriba un programa que permita crear una lista de palabras y que, a continuación, pida una palabra y diga cuántas veces aparece esa palabra en la lista.

Empezar

Solución

                  
 1string palabras[100];
 2string buscar;
 3int cont = 0;
 4int veces = 0;
 5void setup(){
 6    Serial.begin(9600);
 7}
 8void loop(){
 9    Serial.print("Ingrese el número de palabras que desea escribir: ");
10    veces = Serial.read();
11    for(int i=0; i < veces; i++){
12        Serial.print("Ingrese una palabra: ");
13        palabras[i] = Serial.read();
14    }
15    Serial.print("Ingrese la palabra que desea buscar: ");
16    buscar = Serial.read();
17    for(int i=0; i < veces; i++){
18        if(palabras[i] == buscar){
19            cont++;
20        }
21    }
22    if(cont>0){
23        Serial.print("El número de veces que aparece la palabra es: ");
24        Serial.print(cont);
25    }else{
26        Serial.print("La palabra no se encuentra en la lista.");
27    }
28}
